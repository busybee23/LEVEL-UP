### Solution 1 : ( Sliding Window )

```
class Solution {
public:
    int totalFruit(vector<int>& fruits) {
        int left=0,maxLen = 0;
        unordered_map<int,int> mp;
        for(int i=0;i<fruits.size();i++){
            char right = fruits[i];
            mp[fruits[i]]++;
            
            while((int)mp.size()>2){
                mp[fruits[left]]--;
                if(mp[fruits[left]]==0){
                    mp.erase(fruits[left]);
                }
                left++;
            }
            maxLen = max(maxLen,i-left+1);
        }
        return maxLen;
    }
};
```
